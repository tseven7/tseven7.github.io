单源最短路

模板二刷总结

原理

dijkstra(朴素、堆优化)与点关系密切，贪心
bellmanford,spfa与边关系密切，离散数学
不要因为长得像就搞混了

整体的思想

写到变量特别是数组时有整体思想，这比比考虑他的含义层次更高
至少可以发现没有初始化

堆优化dijkstra

入堆时产生冗余数据，但原理上只要做一次

if(st[ver]) continue;
    st[ver]=1;

spfa

多次入队没有意义

if(!st[j])

    st[j]=1;
    q.push(j);

神奇的spfa模板

#include<iostream>
#include<queue>
#include <cstring>
using namespace std;

const int N=2010,M=10010;
int n,m;
int h[N],e[M],w[M],ne[M],idx;
int dist[N], cnt[N];
bool st[N];

void add(int a,int b,int c)
{
    e[idx]=b, w[idx]=c, ne[idx]=h[a], h[a]=idx++;
}

bool spfa()
{
    queue<int> q;
    for(int i=1;i<=n;++i)
    {
        q.push(i);
        st[i]=1;
    }


    while(q.size())
    {
        int t=q.front();
        q.pop();
        st[t]=0;

        if(cnt[t]>=n) return 1;

        for(int i=h[t];i!=-1;i=ne[i])
        {
            int j=e[i];
            if(dist[j]>dist[t]+w[i])
            {
                dist[j]=dist[t]+w[i];
                cnt[j]++;
                if(!st[j])
                {
                    st[j]=1;
                    q.push(j);
                }
            }
        }
    }
    return 0;
}

int main()
{
    memset(h,-1,sizeof h);

    cin>>n>>m;
    while(m--)
    {
        int a,b,c;
        cin>>a>>b>>c;
        add(a,b,c);
    }

    if(spfa()) puts("Yes");
    else puts("No");

    return 0;

}
